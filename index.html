function checkAvailability() { const domainInput = document.getElementById('domainInput'); const domain = domainInput.value; const [name, extension] = domain.split('.'); const apiUrl = `/api/checkDomains/${name}/${extension}`; fetch(apiUrl) .then(response => { if (!response.ok) { throw new Error(`HTTP error! Status: ${response.status}`); } return response.json(); }) .then(data => { const resultsContainer = document.getElementById('results'); resultsContainer.innerHTML = ''; data.results.forEach(result => { const resultElement = document.createElement('p'); resultElement.textContent = `${result.domain}: ${result.isAvailable ? 'Available' : 'Not Available'}`; resultsContainer.appendChild(resultElement); }); }) .catch(error => { console.error('Error fetching data:', error); return error.response.text(); }) .then(text => console.log('Raw response:', text)); }
